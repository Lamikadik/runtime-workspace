# A slightly different configuration for an internal subnet.
subnet 192.168.100.32 netmask 255.255.255.248 { (это данные с калькулятора а именно ip адрксс сети)
range 192.168.100.35 192.168.100.37; (эти данные у нас с калькулятора, там вообще получается ip первого хоста 192.168.100.33, ip последнего хоста 192.168.100.38, но мы не должны брать крайнии)
option domain-name-servers 192.168.100.2; (что бы узнать этот ip адресс мы должны зайти на hq-srv и прописать команду "ip a" там посмотреть на ens33 и ip должен быть 192.168.100.2/28)
option domain-name "au-team.irpo"; (тут меняем имя по заданию, по заданию было au-team.irpo)
option routers 192.168.100.33; (этот ip у нас то же с калькулятора, почему 33? потому что мы когда прописывали пул, а именно 192.168.100.35 192.168.100.37, мы не брали сразу первые значения с калькулятора, а позаботились заранее о ip для роутера) 
option broadcast-address 10.5.5.31; (по заданию широковещательный адрес можно не указывать, так что смело удаляем полностью эту строку)
default-lease-time 600; (это оставялем)
max-lease-time 7200; (это оставялем)
}

Потом меняем данные на нужные значения, должно получится так:

# A slightly different configuration for an internal subnet.
subnet 192.168.100.32 netmask 255.255.255.248 { 
range 192.168.100.35 192.168.100.37;
option domain-name-servers 192.168.100.2; 
option domain-name "au-team.irpo"; (тут меняем имя по заданию, по заданию было au-team.irpo)
option routers 192.168.100.33; 
option broadcast-address 10.5.5.31; (по заданию широковещательный адрес можно не указывать, так что смело удаляем полностью эту строку)
default-lease-time 600; (это оставялем)
max-lease-time 7200; (это оставялем)
}

и всё это сохраняем сочетанием клавиш "ctrl + X" , "Y" , "enter" 

затем пишем команду для запуска службы:

делаем проверку, заходим на cli (клиент), там где параметры сети заходим на ipv4 ставим метод "автоматический", Другие dns-сервера то же удаляем если они были написаны, и жмем "Применить"

options {

	listen-on-v6 port 53 { none; }; (пишем none потому что у нас нет ipv6)
	directory	''var/named''; (это оставялем)
	dump-file	''/var/named/data/cache_dump.db''; (это оставялем)
	statistics-file	''/var/named/data/named_stats.txt''; (это оставялем)
	memstatistics-file ''/var/named/data/named_mem_stats.txt''; (это оставялем)
	secroots-file	''/var/named/data/named.secroots''; (это оставялем)
	recursing-file ''/var/named/data/named.recursing''; (это оставялем)
	allow-query	{ any; }; (тут any для трафика что бы по всей сети прокидовался)
	forwarders { 77.88.8.8; }; (этой записи не было мы ее добавляем, она нужна если запись какая-та не найдётся, то она отправится дальше по сети)
